\begin{frame}

  \vspace{-0.5em}
  \begin{center}
    Checkpoint reason: unconstrained learned routers did not recover oracle modularity.
  \end{center}
\end{frame}

\begin{frame}{Question}
  Previous branch-isolation result:
  \begin{itemize}
    \item separate branch towers alone stayed functionally entangled;
    \item oracle routing made local depend on branch 0 and induction depend on branch 1.
  \end{itemize}

  \vspace{0.8em}
  This checkpoint asks:

  \begin{quote}
    Can a learned router discover role-specific branch modularity without
    hard-coded local-to-branch-0 and induction-to-branch-1 assignment?
  \end{quote}
\end{frame}

\begin{frame}{Setup}
  Same local-vs-induction competition task:
  \begin{itemize}
    \item local copy: \texttt{[x, SEP, x]};
    \item induction continuation: \texttt{[y1,...,y16,y1,...,y16]};
    \item local weight 0.25, induction weight 1.0;
    \item five seeds per config, 1200 training steps.
  \end{itemize}

  \vspace{0.6em}
  Four variants:
  \begin{itemize}
    \item \texttt{branch\_sum}: two branches active everywhere.
    \item \texttt{learned\_position\_router}: branch gate learned per position.
    \item \texttt{learned\_token\_router}: branch gate learned from token/position state.
    \item \texttt{oracle\_route}: local uses branch 0; induction uses branch 1.
  \end{itemize}
\end{frame}

\begin{frame}{Interpretation}
  \textbf{Supported in this toy setup:}
  \begin{itemize}
    \item Learned routers can solve the task.
    \item Gate specialization and causal modularity can dissociate.
    \item Explicit routing can produce branch-level functional modularity.
  \end{itemize}

  \vspace{0.6em}
  \textbf{Not supported by this run:}
  \begin{quote}
    Unconstrained learned routing naturally discovers the role-specific oracle
    split between local and induction.
  \end{quote}
\end{frame}

\begin{frame}{Decision}
  Continue Phase 3, but sharpen the question:

  \begin{quote}
    What routing signal, regularizer, or task pressure is sufficient for learned
    functional modularity to emerge, if any?
  \end{quote}

  \vspace{0.8em}
  Next decisive tests:
  \begin{itemize}
    \item weak router supervision only on scored positions;
    \item entropy or load-balancing pressure;
    \item a more conflict-heavy task where shared-branch solutions are less attractive.
  \end{itemize}
\end{frame}
